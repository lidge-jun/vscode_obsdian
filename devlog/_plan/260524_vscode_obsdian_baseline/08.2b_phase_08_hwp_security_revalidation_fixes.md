```typescript
export async function createSecureRhwpEditor(container: HTMLElement, options: SecureRhwpEditorOptions) {
  const iframe = document.createElement('iframe');
  iframe.src = options.studioUrl;
  iframe.style.width = options.width;
  iframe.style.height = options.height;
  iframe.setAttribute('sandbox', 'allow-scripts allow-same-origin allow-forms allow-downloads');
  container.appendChild(iframe);

  const expectedOrigin = new URL(options.studioUrl).origin;
  const pending = new Map<number, PendingRequest>();
  function onMessage(event: MessageEvent) {
    if (event.source !== iframe.contentWindow) return;
    if (event.origin !== expectedOrigin) return;
    const message = validateRhwpResponse(event.data);
    if (!message) return;
    // resolve/reject pending by id
  }
  window.addEventListener('message', onMessage);

  return {
    loadFile: (data: Uint8Array, fileName: string) => request('loadFile', { data, fileName }),
    exportHwp: () => request('exportHwp'),
    destroy: () => {
      window.removeEventListener('message', onMessage);
      pending.clear();
      iframe.remove();
    },
  };
}
```

Required payloads:

```typescript
export interface HwpFileDataPayload {
  fileName: string;
  buffer: number[];
  fileSize: number;
  isHwpx: boolean;
  error?: string;
}

export interface HwpSavePayload {
  bytes: number[];
}

export interface HwpSaveResultPayload {
  success: boolean;
  savedPath?: string;
  convertedFromHwpx?: boolean;
  error?: string;
}

export function isHwpEvent(type: unknown): type is HwpEvent {
  return typeof type === 'string' && Object.values(HWP_EVENTS).includes(type as HwpEvent);
}

export function validateHwpPayload(type: HwpEvent, payload: unknown): boolean {
  switch (type) {
    case HWP_EVENTS.init:
      return payload === undefined;
    case HWP_EVENTS.fileData:
      return isHwpFileDataPayload(payload);
    case HWP_EVENTS.requestSave:
      return isHwpSavePayload(payload);
    case HWP_EVENTS.saveResult:
      return isHwpSaveResultPayload(payload);
    default:
      return true;
  }
}
```

## 13. Secure rhwpBridge Protocol Details

`createSecureRhwpEditor.ts` must replicate the `@rhwp/editor` protocol with
strict origin targeting.

```typescript
let requestId = 0;

function request(method: string, params = {}) {
  return new Promise((resolve, reject) => {
    const id = ++requestId;
    const timeout = window.setTimeout(() => {
      pending.delete(id);
      reject(new Error(`Request timeout: ${method}`));
    }, options.requestTimeoutMs);
    pending.set(id, { resolve, reject, timeout });
    iframe.contentWindow?.postMessage({ type: 'rhwp-request', id, method, params }, expectedOrigin);
  });
}

async function waitReady() {
  for (let i = 0; i < 30; i++) {
    try {
      if (await request('ready')) return;
    } catch {}
    await new Promise((resolve) => setTimeout(resolve, 500));
  }
  throw new Error('Editor initialization timeout');
}
```

The iframe `load` event must resolve before `waitReady()`. Response validator
must accept only:

```typescript
{ type: 'rhwp-response', id: number, error?: string, result?: unknown }
```
